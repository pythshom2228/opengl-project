cmake_minimum_required(VERSION 3.29)


set(PROJECT_NAME app)
project(${PROJECT_NAME} LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20) 
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "bin")

set(SOURCE "src/main.cpp")

set(SFML_DIR "lib/cmake/SFML")
find_package(SFML 2.6.1 COMPONENTS window REQUIRED)



add_executable(${PROJECT_NAME} ${SOURCE})

file(GLOB DL_LIBS "bin/*.dll")

file(COPY ${DL_LIBS} DESTINATION ${CMAKE_RUNTIME_OUTPUT_DIRECTORY})




if(WIN32)
    set(OPENGL opengl32)
elseif(UNIX)
    set(OPENGL GL)
endif()

target_include_directories(${PROJECT_NAME} PRIVATE "include/GL" "include/glm")

target_link_libraries(${PROJECT_NAME} -L${CMAKE_SOURCE_DIR}/lib stb_image)
target_link_libraries(${PROJECT_NAME} -L${CMAKE_CURRENT_LIST_DIR}/lib glew32)
target_link_libraries(${PROJECT_NAME} sfml-window ${OPENGL})
target_link_libraries(${PROJECT_NAME} stb_image)
